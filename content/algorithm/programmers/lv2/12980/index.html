def solution(n):
    dp = [0 for _ in range(0, n)]
    dp[0], dp[1], dp[2] = 0, 1, 1
    for i in range(3, n):
        if i % 2 == 0:
            dp[i] = dp[int((i + 1) / 2)]
        else:
            dp[i] = dp[i - 1] + 1
        print(i, dp)
    return dp[-1]
