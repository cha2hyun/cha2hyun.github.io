def solution(n):
    fabo = [None] * (n + 1)
    fabo[0] = 0
    fabo[1] = 1
    for i in range(n + 1):
        if i < 2:
            pass
        else:
            fabo[i] = (fabo[i - 1] + fabo[i - 2]) % 1234567
    return fabo[n]

def fibonacci(num):
    a,b = 0,1
    for i in range(num):
        a,b = b,a+b
    return a
