def solution(n):
    answer = 0
    for x in range(1, (n + 1)):
        temp = 0
        for y in range(x, (n + 1) // 2 + 1):
            temp += y
            if temp == n:
                answer += 1
                break
            elif temp > n:
                break
    return answer + 1

def solution(n):
    answer = 0
    for i in range(1, n + 1, 2): # 2만큼 증가하여 짝수는 고려하지 않음
        if n % i == 0:
            answer += 1
    return answer
