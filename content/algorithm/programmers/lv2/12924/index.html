def solution(n):
    answer = 0
    for x in range(1, (n + 1)):
        temp = 0
        for y in range(x, (n + 1) // 2 + 1):
            temp += y
            if temp == n:
                answer += 1
                break
            elif temp > n:
                break
    return answer + 1
