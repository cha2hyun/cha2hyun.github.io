def solution(stones, k):
    # Kë¥¼ ëŒ
    answer = len(stones)
    for i in range(len(stones) - k + 1):
        maximum = 0
        for j in range(i, i + k):
            maximum = max(maximum, stones[j])
        answer = min(answer, maximum)
    return answer

def jump(stones, k, mid):
    cnt = k
    for stone in stones:
        if stone <= mid:
            cnt -= 1
            if cnt == 0:
                return False
        else:
            cnt = k
    return True
