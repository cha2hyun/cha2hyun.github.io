def solution(begin, target, words):
    from collections import deque
    if target not in words:
        return 0

    q = deque()

def isOneAlphabetDiff(prev, next):
    cnt = 0
    for i in range(len(next)):
        if prev[i] != next[i]:
            cnt += 1
        if cnt > 1:
            return False
    return True
