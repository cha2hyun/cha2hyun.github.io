def solution(A, B):
    length = len(A)
    total = 0

    for i in range(length):
        Max = max(A)
        Min = min(B)
        A.remove(Max)
        B.remove(Min)
        total += Max * Min
    return total

    B.sort()
    for i in range(len(A)):

    return total
